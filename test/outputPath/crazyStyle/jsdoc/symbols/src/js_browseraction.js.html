1 /**
  2  * browseraction
  3  * 该文件是浏览器图标js，负责下拉菜单事件绑定。
  4  * @author fedeoo <tragedymi[at]163.com>
  5  * @version 1.0
  6  */
  7 window.onload = function () {
  8     BrowserAction.init();
  9 }
 10 /**
 11  * @global
 12  * @namespace
 13  * @name BrowserAction
 14  */
 15 var BrowserAction = {
 16     options: {
 17         enable: true
 18     },
 19 
 20 
 21     init: function () {
 22         this.bindEvent();
 23     },
 24 
 25     /**
 26      * [bindEvent 绑定下拉事件]
 27      */
 28     bindEvent: function () {
 29         var enableBtn = document.querySelector('.enable');
 30         enableBtn.onclick = this._getEnableClickHandler(enableBtn);
 31 
 32         var foolsBtn = document.querySelector('.foolsBtn');
 33         foolsBtn.onclick = this._getFoolsClickHandler(foolsBtn);
 34     },
 35     /**
 36      * [_getEnableClickHandler 点击处理事件，发送消息给contentscript]
 37      * @param  {[DOM]} enableBtn [页面节点]
 38      */
 39     _getEnableClickHandler: function (enableBtn) {
 40         var bg = chrome.extension.getBackgroundPage(); 
 41         return function () {
 42             if (bg.utilCPU.options.enable === false) {
 43                 bg.utilCPU.options.enable = true;
 44                 enableBtn.classList.add('active');
 45                 bg.utilCPU.turnon();
 46             } else {
 47                 bg.utilCPU.options.enable = false;
 48                 enableBtn.classList.remove('active');
 49                 bg.utilCPU.turnoff();
 50             }
 51         };
 52     },
 53 
 54     /**
 55      * [_getFoolsClickHandler 愚人节点击处理事件，发送消息给contentscript]
 56      * @param  {[DOM]} enableBtn [页面节点]
 57      */
 58     _getFoolsClickHandler: function (foolsBtn) {
 59         var bg = chrome.extension.getBackgroundPage(); 
 60         return function () {
 61             if (foolsBtn.classList.contains('active')) {
 62                 foolsBtn.classList.remove('active');
 63                 bg.utilCPU.sendMessageToCurrentTab({type:'system', cmd:{foolsStyle : 'disable'}});
 64             } else {
 65                 foolsBtn.classList.add('active');
 66                 bg.utilCPU.sendMessageToCurrentTab({type:'system', cmd:{foolsStyle : 'enable'}});
 67             }
 68         };
 69     }
 70 }

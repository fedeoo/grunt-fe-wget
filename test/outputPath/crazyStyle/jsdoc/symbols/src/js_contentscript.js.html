1 /**
  2  * contentscript
  3  * 真正能够操作页面DOM的脚本程序。
  4  * 该脚本监听background的指示。负责往页面插入、移除相应js/css文件。
  5  * 因为代码片段不利管理，采用整个文件的方式。
  6  * @author fedeoo <tragedymi[at]163.com>

  8  */

 10 // console.log('why???');
 11 /* contentscript 在每个页面的独立环境执行 */
 12 // debugger;
 13 var CONFIG = {
 14     CSSMap:{
 15         'globalStyle': 'css/qindeoo.css',
 16         'foolsStyle': 'css/april-fools.css',
 17         'nightSkyStyle': 'css/nightsky.css'
 18     },
 19     JSMap: {
 20         'nightSkyStyle': 'modulejs/nightsky.js'
 21     }
 22 };
 23 /**
 24  * [接收消息并执行对应操作]
 25  * @param  {[type]} request      [消息请求正文]
 26  * @param  {[type]} sender       [description]
 27  * @param  {[type]} sendResponse [description]
 28  */
 29 chrome.extension.onMessage.addListener(function (request, sender, sendResponse) {
 30     //━ ━．━\r．．\r━．
 31     console.debug("━ ━．━\r\t．．\r\t━．");
 32     if (request.type === 'system') {
 33         var cmd = request.cmd;
 34         for (var prop in cmd) {
 35             if (cmd.hasOwnProperty(prop)) {
 36                 if (cmd[prop] === 'enable') {
 37                    CONFIG.CSSMap[prop] && addStyleFile(chrome.extension.getURL(CONFIG.CSSMap[prop]));
 38                    CONFIG.JSMap[prop] && addJsFile(chrome.extension.getURL(CONFIG.JSMap[prop]));
 39                 } else if (cmd[prop] === 'disable') {
 40                     CONFIG.CSSMap[prop] && removeStyleFile(chrome.extension.getURL(CONFIG.CSSMap[prop]));
 41                     CONFIG.JSMap[prop] && removeJsFile(chrome.extension.getURL(CONFIG.JSMap[prop]));
 42                 }
 43             }
 44         }
 45     } else if (request.type === 'user') {
 46         var cmd = request.cmd;
 47         cmd.styleURL && addStyleFile(cmd.styleURL);
 48         cmd.jsURL && addJsFile(cmd.jsURL);
 49     }
 50     
 51 });
 52 
 53 /**
 54  * [addStyleFile 在页面插入css文件，当然也可以是网络资源，如果是扩展资源需要在清单中配置]
 55  * @param {[String]} fileName [文件名称，包括全部路径]
 56  */
 57 function addStyleFile (fileName) {
 58     if (document.head.querySelector('link[href="' + fileName + '"]')) {
 59         return ;
 60     }
 61     var style = document.createElement('link');
 62     style.rel = 'stylesheet';
 63     style.type = 'text/css';
 64     style.href = fileName;
 65     (document.head || document.documentElement).appendChild(style);
 66 }
 67 
 68 /**
 69  * [removeStyleFile 在页面移除之前添加的css文件]
 70  * @param  {[String]} fileName [文件名称，包括全部路径]
 71  */
 72 function removeStyleFile (fileName) {
 73     var style = document.head.querySelector('link[href="' + fileName + '"]');
 74     if (style) {
 75         style.remove();
 76     }
 77 }
 78 
 79 /**
 80  * [addJsFile 在页面插入JS文件]
 81  * @param {[String]} fileName [文件名称，包括全部路径]
 82  */
 83 function addJsFile (fileName) {
 84     if (document.head.querySelector('script[src="' + fileName + '"]')) {
 85         return ;
 86     }
 87     var script = document.createElement("script"); 
 88     script.type = "text/javascript"; 
 89     script.charset = "utf-8";
 90     script.src = fileName; 
 91     (document.head || document.documentElement).appendChild(script); 
 92 }
 93 /**
 94  * [removeJsFile 在页面移除已插入的JS文件]
 95  * @param  {[type]} fileName [文件名称，包括全部路径]
 96  */
 97 function removeJsFile (fileName) {
 98     var script = document.head.querySelector('script[src="' + fileName + '"]');
 99     if (script) {
100         script.remove();
101     }
102 }

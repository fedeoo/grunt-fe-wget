 36     init: function () {
 37         this.reloadContentScript();
 38         this.turnon();
 39         chrome.tabs.onCreated.addListener(this._getTabCreatedHandler());

 43     /**
 44      * [reloadContentScript 当重新载入扩展时，之前打开页面不存在contentScript，需要重新载入]
 45      */
 46     reloadContentScript: function () {
 47         chrome.tabs.query({}, function (tabs) { 
 48             for (var i = tabs.length - 1; i >= 0; i--) {
 49                 chrome.tabs.executeScript(tabs[i].id, { file: "contentscript.js" }, function () {
 50                 });
 51             }
 52         });
 53     },
 54     /**
 55      * [turnon 发送启用美化样式指示给全部页签]
 56      */
 57     turnon: function () {
 58         this.sendMessageToAllTabs({type:'system', cmd:{globalStyle: 'enable'}});
 59     },
 60     /**
 61      * [turnoff 发送关闭美化样式指示给全部页签]
 62      */
 63     turnoff: function () {
 64         this.sendMessageToAllTabs({type:'system', cmd:{globalStyle: 'disable'}});
 65     },

100     /**
101      * [_getTabUpdateHandler 页面URL变化都会触发update]
102      */
103     _getTabUpdateHandler: function () {
104         /**
105          * [更新页签的绑定处理函数]
106          * @param  {[Tab]} tab [页签对象]
107          */
108         return function (tabId, changeInfo, tab) {
109             //将百度 google首页当空白页处理
110             if (tab.url === 'http://www.baidu.com/' || tab.url === 'http://www.google.com/'
111                 || tab.url === 'https://www.google.com/' || tab.url === 'https://www.google.com.hk/') {
112                 var now  = new Date();
113                 if (now.getHours() >= 20 || now.getHours() <= 5) {//night
114                     chrome.tabs.sendMessage(tab.id, { type:'system', cmd:{nightSkyStyle: 'enable'}}, function (response) {
115                     // console.log('add Style success in tab');
116                     });
117                 }
118             }
119             if (this.options.enable === true) {
120                 // chrome.tabs.executeScript(tabId, { file: "contentscript.js" }, function () {
121                 // });
122                 chrome.tabs.sendMessage(tabId, {type: 'system', cmd:{globalStyle: 'enable'}}, function (response) {
123                     // console.log('add Style success in tab');
124                 });
125             }
126 
127             var url = tab.url;
128             if (settings.options) {
129                 var options = settings.options;
130                 for (var i = options.length - 1; i >= 0; i--) {
131                     if (url.indexOf(options[i].match) !== -1) {
132                         chrome.tabs.sendMessage(tabId, {type:'user', cmd:options[i]}, function (argument) {
133                             // 
134                         })
135                     }
136                 };
137             }
138 
139         };
140     },

142     /**
143      * [sendMessageToAllTabs 给所有标签发送消息]
144      * @param  {[any]} message [消息内容]
145      */
146     sendMessageToAllTabs: function (message) {
147         chrome.tabs.query({}, function (tabs) { 
148             for (var i = tabs.length - 1; i >= 0; i--) {
149                 chrome.tabs.sendMessage(tabs[i].id, message, function (response) {
150                 });
151             }
152         });
153     },
154 
155     /**
156      * [sendMessageToCurrentTab 发送到当前激活页签，多个窗口可能有多个激活页签]
157      * @param  {[any]} message [消息内容]
158      */
159     sendMessageToCurrentTab: function (message) {
160         chrome.tabs.query({ currentWindow: true, active: true }, function (tabs) {

162                 chrome.tabs.sendMessage(tabs[i].id, message, function (response) {
163                 });
